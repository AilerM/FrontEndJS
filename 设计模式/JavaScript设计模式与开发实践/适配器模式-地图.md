# 适配器模式

适配器的别名是包装器(wrapper)，这是一个相对简单的模式

```js
var googleMap = { show: function(){
  console.log( '开始渲染谷歌地图' ); }
};
var baiduMap = {
  display: function(){
  console.log( '开始渲染百度地图' ); }
};
var baiduMapAdapter = { 
  show: function(){
    return baiduMap.display();
  }
}

renderMap( googleMap );
renderMap( baiduMapAdapter );
```
